After downloading the files and running the string on the terminal, I got a massive string of text, which I didn't know what to do with.

I slowly started to reverse the working of the encrypting done by the scrambler to get this code:
```Python
def unscramble(L):
    A = L[:]
    i = len(A) - 1    
    while i >= 2:
        if A[i] and isinstance(A[i][-1], list):
            appended_slice = A[i].pop()
            slice_length = len(appended_slice)
        else:
            if A[i]:
                A[i].pop()  
            slice_length = 0
        if len(A[i-1]) > slice_length:
            popped_element = A[i-1][slice_length:]
            A[i-1] = A[i-1][:slice_length]
            A.insert(i, popped_element)        
        i -= 1    
    return A

def reverse(h):
    chars = []
    for item in h:
        if item and len(item) > 0:
            first_element = item[0]
            if isinstance(first_element, str) and first_element.startswith('0x'):
                hex_str = first_element.strip().strip('"').strip("'")
                chars.append(chr(int(hex_str, 16)))
    return ''.join(chars)
```
